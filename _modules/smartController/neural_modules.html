class RecurrentModel(nn.Module):
    """
    Implements a recurrent neural network using GRU layers. This is used to process sequential data.
    
    Args:
        input_size (int): Number of input features.
        hidden_size (int): Number of hidden units in the GRU.
        device (str): Device to run the model ('cpu' or 'cuda').
    """

    def __init__(self, input_size, hidden_size, device='cpu'):
        super(RecurrentModel, self).__init__()
        self.device = device
        self.hidden_size = hidden_size
        self.gru = nn.GRU(input_size, hidden_size, batch_first=True)



[docs]
    def forward(self, x):
        """
        Forward pass through the GRU layer.
        
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, seq_length, input_size).
        
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, hidden_size).
        """
        h0 = torch.zeros(1, x.size(0), self.hidden_size).to(x.device)
        out, _ = self.gru(x, h0)
        return F.relu(out[:, -1, :])

    def get_oh_labels(self, decimal_labels, n_way):
        """
        Convert labels to one-hot encoding.
        
        Args:
            decimal_labels (torch.Tensor): Class labels in decimal format.
            n_way (int): Number of classes.
        
        Returns:
            torch.Tensor: One-hot encoded labels.
        """
        labels_onehot = torch.zeros([decimal_labels.size()[0], n_way], device=self.device)
        labels_onehot = labels_onehot.scatter(1, decimal_labels, 1)
        return labels_onehot

    def get_centroids(self, hidden_vectors, onehot_labels):
        """
        Calculate class centroids from support set latent vectors.
        
        Args:
            hidden_vectors (torch.Tensor): Latent vectors of the support set.
            onehot_labels (torch.Tensor): One-hot encoded class labels.
        
        Returns:
            torch.Tensor: Centroids of each class.
            torch.Tensor: Boolean mask indicating missing clusters.
        """
        cluster_agg = onehot_labels.T @ hidden_vectors
        samples_per_cluster = onehot_labels.sum(0)
        centroids = torch.zeros_like(cluster_agg, device=self.device)
        missing_clusters = samples_per_cluster == 0
        existent_centroids = cluster_agg[~missing_clusters] / samples_per_cluster[~missing_clusters].unsqueeze(-1)
        centroids[~missing_clusters] = existent_centroids
        return centroids, missing_clusters

    def forward(self, hidden_vectors, labels, known_attacks_count, query_mask):
        """
        Perform forward pass for classification.
        
        Args:
            hidden_vectors (torch.Tensor): Latent vectors of the input data.
            labels (torch.Tensor): True class labels.
            known_attacks_count (int): Number of known attack classes.
            query_mask (torch.Tensor): Boolean mask indicating which samples are queries.
        
        Returns:
            torch.Tensor: Scores for each query sample.
        """
        oh_labels = self.get_oh_labels(decimal_labels=labels.long(), n_way=known_attacks_count)
        centroids, _ = self.get_centroids(hidden_vectors[~query_mask], oh_labels[~query_mask])
        scores = 1 / (torch.cdist(hidden_vectors[query_mask], centroids) + 1e-10)
        return scores

    def forward(self, scores):
        """
        Forward pass that computes confidence scores.
        
        Args:
            scores (torch.Tensor): Prediction scores.
        
        Returns:
            torch.Tensor: Confidence indicators.
        """
        scores = (1 - scores.unsqueeze(-1)).min(1)[0]
        unknown_indicators = torch.sigmoid(scores)
        return unknown_indicators

class KernelRegressionLoss(nn.Module):
    """
    Implements the kernel regression loss function, including repulsive and attractive forces.
    
    Args:
        repulsive_weight (int): Weight for the repulsive force in the loss function.
        attractive_weight (int): Weight for the attractive force in the loss function.
        device (str): Device to run the model ('cpu' or 'cuda').
    """

    def __init__(self, repulsive_weight=1, attractive_weight=1, device='cpu'):
        super(KernelRegressionLoss, self).__init__()
        self.r_w = repulsive_weight
        self.a_w = attractive_weight
        self.device = device



[docs]
    def forward(self, baseline_kernel, predicted_kernel):
        """
        Compute the kernel regression loss.
        
        Args:
            baseline_kernel (torch.Tensor): Baseline kernel matrix.
            predicted_kernel (torch.Tensor): Predicted kernel matrix.
        
        Returns:
            torch.Tensor: Loss value.
        """
        repulsive_CE_term = -(1 - baseline_kernel) * torch.log(1 - predicted_kernel + 1e-10)
        repulsive_CE_term = repulsive_CE_term.sum(dim=1).mean()

        attractive_CE_term = -(baseline_kernel * torch.log(predicted_kernel + 1e-10))
        attractive_CE_term = attractive_CE_term.sum(dim=1).mean()

        return (self.r_w * repulsive_CE_term) + (self.a_w * attractive_CE_term)

class SimmilarityNet(nn.Module):
    """
    A neural network that computes the similarity between two vectors.
    
    Args:
        h_dim (int): Dimensionality of the input hidden vectors.
    """

    def __init__(self, h_dim):
        super(SimmilarityNet, self).__init__()
        self.act = nn.LeakyReLU(0.2)
        self.fc1 = nn.Linear(h_dim, h_dim // 2)
        self.fc2 = nn.Linear(h_dim // 2, 1)



[docs]
    def forward(self, x1, x2):
        """
        Compute the similarity between two input vectors.
        
        Args:
            x1 (torch.Tensor): First input vector.
            x2 (torch.Tensor): Second input vector.
        
        Returns:
            torch.Tensor: Similarity score between the two vectors.
        """
        input_to_symm = torch.abs(x1 - x2)
        symm = self.fc1(input_to_symm)
        symm = self.act(symm)
        symm = self.fc2(symm)
        return symm

class HighDimKernelRegressor(nn.Module):
    """
    A high-dimensional kernel regressor for computing pairwise similarity between latent vectors.
    
    Args:
        in_features (int): Number of input features.
        out_features (int): Number of output features.
        n_heads (int): Number of attention heads.
        device (str): Device to run the model ('cpu' or 'cuda').
    """

    def __init__(self, in_features, out_features, n_heads, is_concat=False, dropout=0.0, leaky_relu_negative_slope=0.2, share_weights=True, device='cpu'):
        super(HighDimKernelRegressor, self).__init__()
        self.device = device
        self.w = nn.Parameter(torch.tensor(1.0))
        self.b = nn.Parameter(torch.tensor(-0.5))
        self.similarity_network = SimmilarityNet(h_dim=in_features)

    def forward(self, hiddens):
        """
        Forward pass for kernel regression.
        
        Args:
            hiddens (torch.Tensor): Latent hidden representations.
        
        Returns:
            tuple: (hidden representations, kernel matrix).
        """
        n_nodes = hiddens.shape[0]
        h_pivot = hiddens.repeat(n_nodes, 1)
        h_interleave = hiddens.repeat_interleave(n_nodes, dim=0)

        energies = self.similarity_network(h_pivot, h_interleave)
        kernel = torch.sigmoid(energies)
        kernel = kernel.reshape(n_nodes, n_nodes)

        return hiddens, kernel
